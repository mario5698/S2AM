create table [Users] (
Id int not null,
Name nvarchar(100) not null,
Token  nvarchar(100),
Direction nvarchar(100) not null,
Mobile int not null,
primary key (Id),
)

create table Riders(
Id int not null,
Name nvarchar(100) not null,
Mobile int not null,
Vehicle nvarchar(100) not null,
primary key (Id),
)


create table Tasks (
Id int not null,
Content nvarchar(100) not null,
Direction_Delivery nvarchar(100) not null,
Direction_Pickup nvarchar(100) not null,
UserID int not null,
RiderID int,
Asigned bit,
[State] nvarchar(100)
primary key (id)
foreign key(UserID) references [Users](Id),
foreign key(RiderID) references Riders(Id),
)
