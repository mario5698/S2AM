create table [Users] (
Id int IDENTITY(1,1) not null ,
Name nvarchar(100),
Token  nvarchar(100),
Direction nvarchar(100),
Mobile int,
primary key (Id)
)

create table Riders(
Id int IDENTITY(1,1) not null,
Name nvarchar(100),
Mobile int,
Vehicle nvarchar(100),
primary key (Id)
)


create table Tasks (
Id int IDENTITY(1,1) not null,
Content nvarchar(100),
Direction_Delivery nvarchar(100),
Direction_Pickup nvarchar(100),
UserID int not null,
RiderID int,
Asigned bit,
[State] nvarchar(100)
primary key (Id),
foreign key(UserID) references [Users](Id),
foreign key(RiderID) references Riders(Id),
)
